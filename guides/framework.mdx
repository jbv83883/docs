<Steps>

  <Step title="Install Visual C++ Redistributables">

    - Download the Visual C++ Redistributable Runtimes All-in-One from TechPowerUp.
    - Open the zip file and run "Install All" as administrator.
    - Allow all runtimes to install.

    <CardGroup cols={1}>
    <Card
        title="Download Visual C++ Redistributables"
        icon="link"
        href="https://www.techpowerup.com/download/visual-c-redistributable-runtime-package-all-in-one/"
    >
    </Card>
    </CardGroup>

  </Step>

  <Step title="Install .NET Framework">

    - Download and install the .NET Framework from the official Microsoft website.
    - Restart your PC after installation.

    <CardGroup cols={1}>
    <Card
        title="Download .NET Framework"
        icon="link"
        href="https://dotnet.microsoft.com/download/dotnet-framework"
    >
    </Card>
    </CardGroup>

  </Step>

  <Step title="Install DirectX End-User Runtime">

    - Download and install the .NET Framework from the official Microsoft website.
    - Restart your PC after installation.

    <CardGroup cols={1}>
    <Card
        title="Download DirectX End-User Runtime"
        icon="link"
        href="https://www.microsoft.com/en-gb/download/details.aspx?id=35"
    >
    </Card>
    </CardGroup>
  </Step>


</Steps>
